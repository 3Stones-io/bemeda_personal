Scenario Event Bridge

Use Phoenix PubSub to bridge scenario events with LiveView real-time updates.

# lib/bemeda_personal/use/scenario_bridge.ex defmodule BemedaPersonal.USE.ScenarioBridge do alias Phoenix.PubSub def broadcast_scenario_event(scenario_id, event, data) do PubSub.broadcast( BemedaPersonal.PubSub, "use_scenario:#{scenario_id}", {event, data} ) end end

# test/use_scenarios/business_scenarios_test.exs defmodule BemedaPersonal.USEScenarios.BusinessScenariosTest do use BemedaPersonal.DataCase use BemedaPersonalWeb.ConnCase describe "B_S001: Cold Call to Placement" do test "US001: Organisation Receives Cold Call", %{conn: conn} do # Given a Healthcare Organisation exists org = create_test_organisation() # When the Sales Team makes contact response = post(conn, "/api/use/scenarios/B_S001_US001/execute") # Then the organisation should be engaged assert json_response(response, 200)["status"] == "success" end end end
